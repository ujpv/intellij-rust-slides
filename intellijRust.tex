\frametitle{Упрощённое Psi-дерево}

\begin{tikzpicture}[
	level distance = 4em,
	every node/.style = {
		shape=rectangle, rounded corners,
		draw, align=center,
		top color=white, bottom color=blue!20}]

	\Tree 
	[.PsiFile 
		[.{ModItem\\mod m1 \{...\}} [.{EnumItem\\enum Action \{...\}} ... ]]
		[.{FnItem\\fn get\_action() \{...\}} ... ]
		[.{FnItem\\fn main() \{...\}} 
			[.{UseItem\\use m1::Action;} ... ]
			[.{CallExpr\\get\_action();} ... ]
		]
	]
\end{tikzpicture}
	
\end{frame}
